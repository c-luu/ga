	\section{Recurrence}
		\begin{equation}
			T(i)=\begin{cases}
				0, & \text{if $|a|=0$}.\\
				dict(a_i), & \text{if $i=0$}.\\
				\underset{1 \leq j \leq i}{\bigvee} dict(a_{j,i+1}) \wedge T(j-1).
			\end{cases}
		\end{equation}
	\section{Assumptions}
	Half-open intervals for sequence ranges, i.e. $[i,j)$.\\
	0-based indexing.
